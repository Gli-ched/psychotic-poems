<script>
let lastSpawn = 0;
document.addEventListener('mousemove', (e) => {
  const now = Date.now();
  if (now - lastSpawn > 200) {  // one glow every 200ms
    lastSpawn = now;

    const drop = document.createElement('div');
    drop.className = 'blood-drop';
    drop.style.left = e.clientX + 'px';
    drop.style.top = e.clientY + 'px';
    document.body.appendChild(drop);

    // After fade → spawn soul
    setTimeout(() => {
      drop.remove();

      const soul = document.createElement('div');
      soul.className = 'soul-escape';
      soul.style.left = e.clientX + 'px';
      soul.style.top = e.clientY + 'px';
      document.body.appendChild(soul);

      setTimeout(() => {
        soul.remove();
      }, 2000); // remove soul gif
    }, 1500); // glow lives 1.5s
  }
});

function scrollToPoem() {
    const selectedPoem = document.getElementById('poem-select').value;
    document.querySelector(selectedPoem).scrollIntoView({ behavior: 'smooth' });
}


const music = document.getElementById("bg-music");
  let canPlay = true; // flag to check if music is ready to play again

  document.addEventListener("click", () => {
    if (canPlay) {
      music.currentTime = 0; // start from beginning
      music.play();
      canPlay = false; // prevent overlapping
    }
  });

  // when audio finishes, allow it to play again on next click
  music.addEventListener("ended", () => {
    canPlay = true;
  });

</script>
